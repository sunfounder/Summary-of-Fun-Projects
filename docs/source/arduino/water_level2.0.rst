**Common Connections:**

* **Water Level Detection Module**

  - **A:** Connect to **A0** on the Arduino.
  - **G:** Connect to breadboard‚Äôs negative power bus.
  - **V:** Connect to breadboard‚Äôs red power bus.

* **LEDS**

  - Connect the LED **cathode** to a **1KŒ© resistor** then to the negative power bus on the breadboard, and the **anode** to the **2~9** on the Arduino.

.. code-block:: arduino

      // LED pin array
      const int ledPins[] = {2, 3, 4, 5, 6, 7, 8, 9};
      const int numLEDs = 8;

      // Water level sensor analog input pin
      const int sensorPin = A0;

      void setup() {
        Serial.begin(9600);
        // Set all LED pins as output
        for (int i = 0; i < numLEDs; i++) {
          pinMode(ledPins[i], OUTPUT);
        }
      }

      void loop() {
        // Read the water level sensor value (0~1023)
        int sensorValue = analogRead(sensorPin);
        Serial.print("Water level analog value: ");
        Serial.println(sensorValue);

        // Map the analog value to number of LEDs (0~8)
        int level = map(sensorValue, 0, 450, 0, numLEDs);

        // Turn on LEDs according to the water level
        for (int i = 0; i < numLEDs; i++) {
          if (i < level) {
            digitalWrite(ledPins[i], HIGH);
          } else {
            digitalWrite(ledPins[i], LOW);
          }
        }

        delay(500);  // Delay before next update
      }
